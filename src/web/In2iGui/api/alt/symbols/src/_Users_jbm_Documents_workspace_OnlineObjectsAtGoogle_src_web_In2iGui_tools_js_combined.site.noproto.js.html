404 n2i.animation.render = function(element) {
405 	this.running = true;
406 	var next = false;
407 	var stamp = new Date().getTime();
408 	for (id in this.objects) {
409 		var obj = this.objects[id];
410 		if (obj.work) {
411 			for (var i=0; i < obj.work.length; i++) {
412 				var work = obj.work[i];
413 				if (work.finished) continue;
414 				var place = (stamp-work.start)/(work.end-work.start);
415 				if (place<0) {
416 					next=true;
417 					continue;
418 				}
419 				else if (isNaN(place)) place = 1;
420 				else if (place>1) place=1;
421 				else if (place<1) next=true;
422 				var v = place;
423 				if (work.delegate && work.delegate.ease) {
424 					v = work.delegate.ease(v);
425 				}
426 				var value = null;
427 				if (!work.css) {
428 					obj.element[work.property] = Math.round(work.from+(work.to-work.from)*v);
429 				} else if (work.to.red!=null) {
430 					var red = Math.round(work.from.red+(work.to.red-work.from.red)*v);
431 					var green = Math.round(work.from.green+(work.to.green-work.from.green)*v);
432 					var blue = Math.round(work.from.blue+(work.to.blue-work.from.blue)*v);
433 					value = 'rgb('+red+','+green+','+blue+')';
434 					obj.element.style[work.property]=value;
435 				} else if (n2i.browser.msie && work.property=='opacity') {
436 					var opacity = (work.from+(work.to-work.from)*v);
437 					if (opacity==1) {
438 						obj.element.style['filter']='';
439 					} else {
440 						obj.element.style['filter']='alpha(opacity='+(opacity*100)+')';
441 					}
442 				} else {
443 					value = new String(work.from+(work.to-work.from)*v)+(work.unit!=null ? work.unit : '');
444 					obj.element.style[work.property]=value;
445 				}
446 				if (place==1) {
447 					work.finished = true;
448 					if (work.delegate && work.delegate.onComplete) {
449 						work.delegate.onComplete();
450 					} else if (work.delegate && work.delegate.hideOnComplete) {
451 						obj.element.style.display='none';
452 					}
453 				}
454 			};
455 		}
456 	}
457 	if (next) {
458 		window.setTimeout(function() {
459 			n2i.animation.render();
460 		},0);
461 	} else {
462 		this.running = false;
463 	}
464 	//window.status = this.running;
